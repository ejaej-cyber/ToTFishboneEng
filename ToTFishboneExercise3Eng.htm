<script>

const data = {

  "Legal Frameworks": ["Weak enforcement","Gaps in legislation","Inconsistent regulations"],

  "Institutional Capacity": ["Limited expertise","Understaffed agencies","Weak coordination"],

  "Infrastructure": ["Lack of treatment facilities","Poor landfill systems","Limited recycling"],

  "Transboundary Movement": ["Illegal dumping","Mislabeling shipments","Weak border monitoring"],

  "Economic Factors": ["High disposal cost","Limited funding","Informal sector reliance"],

  "Technology & Data": ["Poor data collection","Outdated equipment","Lack of tracking systems"],

  "Public Awareness": ["Low awareness","Informal recycling","Limited education"],

  "Environment": ["Hurricane disruptions","Limited land space","Coastal contamination"]

};

 

let selectedBones = [];

let items = [];

let placements = {};

let dragged = null;

 

function shuffle(arr){ return arr.sort(()=>Math.random()-0.5); }

 

// create buttons

Object.keys(data).forEach(bone=>{

  const btn=document.createElement("button");

  btn.innerText=bone;

  btn.onclick=()=>{

    if(selectedBones.includes(bone)){

      selectedBones=selectedBones.filter(b=>b!==bone);

      btn.classList.remove("selected");

    } else {

      if(selectedBones.length>=6) return alert("Max 6 bones");

      selectedBones.push(bone);

      btn.classList.add("selected");

    }

    load(); draw();

  };

  bones.appendChild(btn);

});

 

function load(){

  subtopics.innerHTML="";

  items=shuffle(selectedBones.flatMap(b=>data[b].map(s=>({text:s,bone:b}))));

  placements={};

 

  items.forEach(item=>{

    const el=document.createElement("div");

    el.innerText=item.text;

 

    // desktop

    el.draggable=true;

    el.ondragstart=()=>{ dragged=item; el.classList.add("dragging"); };

    el.ondragend=()=>el.classList.remove("dragging");

 

    // mobile

    el.ontouchstart=()=>{ dragged=item; el.classList.add("dragging"); };

 

    subtopics.appendChild(el);

  });

}

 

function draw(){

  diagram.innerHTML="";

 

  selectedBones.forEach((bone,i)=>{

    const div=document.createElement("div");

    div.className="bone";

 

    const top=i%2===0;

    div.style.left=(10+i*12)+"%";

    div.style.top=top?"20%":"70%";

    div.style.transform=top?"rotate(-40deg)":"rotate(40deg)";

 

    div.ondragover=e=>e.preventDefault();

    div.ondrop=()=>place(bone);

 

    div.ontouchend=()=>place(bone);

 

    const line=document.createElement("div"); line.className="line";

    const label=document.createElement("div"); label.innerText=bone;

    const drop=document.createElement("div"); drop.className="dropzone";

 

    Object.keys(placements).forEach(t=>{

      if(placements[t]===bone){

        const d=document.createElement("div");

        d.innerText=t;

        drop.appendChild(d);

      }

    });

 

    div.appendChild(line);

    div.appendChild(label);

    div.appendChild(drop);

    diagram.appendChild(div);

  });

}

 

function place(bone){

  if(!dragged) return;

  placements[dragged.text]=bone;

  dragged=null;

  draw();

}

 

function checkAnswers(){

  document.querySelectorAll("#subtopics div").forEach(el=>{

    const item=items.find(i=>i.text===el.innerText);

    if(placements[item.text]===item.bone){

      el.classList.add("correct");

    } else {

      el.classList.add("incorrect");

    }

  });

}
